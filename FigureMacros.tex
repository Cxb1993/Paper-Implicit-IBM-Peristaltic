% Commands for including a standard figure
\newcommand{\singlefigure}[1]{
	\includegraphics[bb=.1in .1in 7.45in 5.85in,width=3.9in,clip]{Images/#1}
}
\newcommand{\singlefigureall}[3]{
\begin{figure}
\begin{center}
\singlefigure{#1}
\end{center}
\vspace{-.2 in}
\caption{\small #2}
\label{fig:#3}
\end{figure}
\vspace{-.285 in}
}

\newcommand{\halffigure}[1]{

}

\doubletopspace

\end{center}
\doublebottomspace
}
